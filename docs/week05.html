HW01_ObsPy.py 
from obspy.clients.fdsn import Client
from obspy import UTCDateTime

client = Client("IRIS")
starttime = UTCDateTime("2025-10-07T23:52:12")
st = client.get_waveforms("TW", "NACB", "", "BHZ", starttime, starttime + 120)

st.plot(outfile="waveform.png")
# 地震波圖已成功抓取並儲存
